Quick Start 
from openai import OpenAI

client = OpenAI(
    base_url="http://localhost:8080/v1",
    api_key="aeth-your-key"  # or any string if auth is off
)

response = client.chat.completions.create(
    model="gemini-2.0-flash",  # any model from /v1/models
    messages=[
        {"role": "user", "content": "Hello from my app!"}
    ]
)

print(response.choices[0].message.content)
import OpenAI from "openai";

const client = new OpenAI({
  baseURL: "http://localhost:8080/v1",
  apiKey: "aeth-your-key", // or any string if auth is off
});

const response = await client.chat.completions.create({
  model: "gemini-2.0-flash",
  messages: [{ role: "user", content: "Hello from my app!" }],
});

console.log(response.choices[0].message.content);
curl http://localhost:8080/v1/chat/completions \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer aeth-your-key" \
  -d '{
    "model": "gemini-2.0-flash",
    "messages": [
      {"role": "user", "content": "Hello from my app!"}
    ]
  }'
